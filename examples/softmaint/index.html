  require([
    'd3',
    'js/core',
    'layouts/force',
    'layouts/zoom'
  ], function(
    d3,
    graphyte,
    forceLayout,
    zoomLayout
  ) {
    d3.text('data.gv', function(error, result) {
      if (error) throw error;

      var force = forceLayout();
      var zoom = zoomLayout();

      var graph = graphyte()
        .arrows(true)
        .layout(zoom, force);

      graph.import(result, 'dot');
      d3.select('#graph').call(graph);
    });
  });
